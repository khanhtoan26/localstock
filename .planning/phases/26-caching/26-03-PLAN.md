    ```python
    """Phase 26 / CACHE-02 — PipelineRunRepository (D-01)."""
    from __future__ import annotations
    from sqlalchemy import select
    from sqlalchemy.ext.asyncio import AsyncSession

    from localstock.db.models import PipelineRun


    class PipelineRunRepository:
        """Read-side queries against `pipeline_runs`. Writes still go through
        Pipeline.run_full / AutomationService.run_daily_pipeline."""

        def __init__(self, session: AsyncSession) -> None:
            self.session = session

        async def get_latest_completed(self) -> int | None:
            """Integer PK of the most recent completed run, or None.

            Used by cache key composers (D-01) to version scoring/ranking
            outputs. Wrapped by `cache.version.resolve_latest_run_id` with
            a 5s TTL to avoid hot-loop DB hits (D-02).
            """
            stmt = (
                select(PipelineRun.id)
                .where(PipelineRun.status == "completed")
                .order_by(PipelineRun.completed_at.desc())
                .limit(1)
            )
            result = await self.session.execute(stmt)
            return result.scalar_one_or_none()
    ```

    ```python
    """Phase 26 / D-01 — Version-key resolver.

    `resolve_latest_run_id` returns the integer pipeline_run_id used to
    compose cache keys (e.g. `scores:ranking:run={id}`). The result is
    cached for 5s (D-02 `pipeline:latest_run_id` namespace) to avoid
    hammering the DB on every request.

    Pipeline finalize hooks (26-05) MUST invalidate this namespace so
    readers immediately see the new run_id; otherwise the 5s TTL would
    delay version bumps by up to 5s.
    """
    from __future__ import annotations
    from typing import Awaitable, Callable

    from sqlalchemy.ext.asyncio import AsyncSession

    from localstock.cache import get_or_compute
    from localstock.db.repositories.pipeline_run_repo import PipelineRunRepository


    async def resolve_latest_run_id(
        session_factory: Callable[[], "AsyncSession"],
    ) -> int | None:
        """Return current `pipeline_run_id` for cache-key composition.

        `session_factory` is a zero-arg callable returning an `AsyncSession`
        context manager (matches `AutomationService.session_factory`).
        """

        async def _fetch() -> int | None:
            async with session_factory() as session:
                repo = PipelineRunRepository(session)
                return await repo.get_latest_completed()

        return await get_or_compute(
            namespace="pipeline:latest_run_id",
            key="current",
            compute_fn=_fetch,
        )
    ```

    ```python
    """Closes ROADMAP SC #2 verbatim — pipeline_run_id versioning."""
    import pytest
    from contextlib import asynccontextmanager
    from datetime import datetime, timezone, timedelta

    from localstock.cache import (
        get_or_compute, invalidate_namespace, resolve_latest_run_id,
    )
    from localstock.db.models import PipelineRun


    def _make_factory(db_session):
        @asynccontextmanager
        async def factory():
            yield db_session
        return factory


    @pytest.mark.asyncio
    @pytest.mark.requires_pg
    async def test_new_pipeline_run_invalidates_old_keys(db_session):
        now = datetime.now(timezone.utc)
        # Seed: completed run id=N1
        run1 = PipelineRun(started_at=now - timedelta(hours=1),
                           completed_at=now - timedelta(hours=1),
                           status="completed", run_type="daily")
        db_session.add(run1)
        await db_session.flush()
        factory = _make_factory(db_session)

        # Resolve initial run_id
        rid_a = await resolve_latest_run_id(factory)
        assert rid_a == run1.id

        # Populate scoring cache under that key
        async def compute_old(): return {"snapshot": "from_run_1"}
        v_old = await get_or_compute(
            "scores:ranking", f"limit=50:run={rid_a}", compute_old,
        )
        assert v_old == {"snapshot": "from_run_1"}

        # Insert NEW completed run
        run2 = PipelineRun(started_at=now, completed_at=now,
                           status="completed", run_type="daily")
        db_session.add(run2)
        await db_session.flush()

        # Pipeline finalize (26-05) calls these. Simulate here:
        invalidate_namespace("pipeline:latest_run_id")
        invalidate_namespace("scores:ranking")

        rid_b = await resolve_latest_run_id(factory)
        assert rid_b == run2.id, "version key did not advance after new run"
        assert rid_b != rid_a

        # Composing a key with the NEW run_id must miss (cache cleared)
        async def compute_new(): return {"snapshot": "from_run_2"}
        v_new = await get_or_compute(
            "scores:ranking", f"limit=50:run={rid_b}", compute_new,
        )
        assert v_new == {"snapshot": "from_run_2"}
        # The old key WOULD still hold its value (TTL hasn't expired) but
        # composers will never address it because rid_a is no longer current.
        # That is the SC #2 contract: "key cũ không bao giờ trả stale data".

    @pytest.mark.asyncio
    @pytest.mark.requires_pg
    async def test_resolve_uses_5s_cache(db_session):
        """5s TTL cache on pipeline:latest_run_id (D-02)."""
        now = datetime.now(timezone.utc)
        run = PipelineRun(started_at=now, completed_at=now,
                          status="completed", run_type="daily")
        db_session.add(run); await db_session.flush()
        factory = _make_factory(db_session)

        from localstock.observability.metrics import get_metrics
        m = get_metrics()
        # First call: miss
        before_misses = m["cache_misses_total"].labels(
            cache_name="pipeline:latest_run_id"
        )._value.get()
        await resolve_latest_run_id(factory)
        # Second call within TTL: hit
        before_hits = m["cache_hits_total"].labels(
            cache_name="pipeline:latest_run_id"
        )._value.get()
        await resolve_latest_run_id(factory)
        after_hits = m["cache_hits_total"].labels(
            cache_name="pipeline:latest_run_id"
        )._value.get()
        assert after_hits - before_hits == 1
    ```
