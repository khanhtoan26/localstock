    """Phase 26 / D-08 — X-Cache: hit|miss response header (P-4)."""
    from __future__ import annotations
    from starlette.middleware.base import BaseHTTPMiddleware
    from starlette.requests import Request

    from localstock.cache._context import cache_outcome_var


    class CacheHeaderMiddleware(BaseHTTPMiddleware):
        """Sets `X-Cache: hit | miss` on responses whose handler called
        `get_or_compute(...)`. Routes that did NOT touch the cache emit no
        header (outcome stays None).

        Mirrors `CorrelationIdMiddleware` shape but inverse direction:
        * Reset ContextVar to None BEFORE call_next.
        * Read AFTER. Set header only if non-None.

        See RESEARCH §7 P-4 for the boundary rationale.
        """

        async def dispatch(self, request: Request, call_next):
            token = cache_outcome_var.set(None)
            try:
                response = await call_next(request)
                outcome = cache_outcome_var.get()
                if outcome is not None:
                    response.headers["X-Cache"] = outcome
                return response
            finally:
                cache_outcome_var.reset(token)
    ```

    import pytest
    from fastapi import FastAPI
    from fastapi.testclient import TestClient

    from localstock.cache import cache_outcome_var, get_or_compute
    from localstock.cache.middleware import CacheHeaderMiddleware


    def _build_app():
        app = FastAPI()
        app.add_middleware(CacheHeaderMiddleware)

        @app.get("/cached")
        async def cached_endpoint():
            async def compute(): return {"ok": True}
            return await get_or_compute("scores:ranking", "test", compute)

        @app.get("/uncached")
        async def uncached_endpoint():
            return {"ok": True}
        return app


    def test_cached_route_emits_miss_then_hit():
        client = TestClient(_build_app())
        r1 = client.get("/cached")
        assert r1.status_code == 200
        assert r1.headers.get("X-Cache") == "miss"
        r2 = client.get("/cached")
        assert r2.headers.get("X-Cache") == "hit"


    def test_uncached_route_emits_no_header():
        client = TestClient(_build_app())
        r = client.get("/uncached")
        assert r.status_code == 200
        assert "X-Cache" not in r.headers


    def test_contextvar_reset_between_requests():
        """P-4 — middleware resets ContextVar so a non-caching request after
        a caching one does not inherit a stale outcome."""
        client = TestClient(_build_app())
        client.get("/cached")
        r = client.get("/uncached")
        assert "X-Cache" not in r.headers
    ```
