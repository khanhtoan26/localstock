    ```python
    """Phase 26 / D-05 — Pre-warm hot keys at end of run_daily_pipeline.

    Q-3 — pre-warm goes through the SAME `get_or_compute` choke point that
    routes use. No direct cache writes; no double-compute hazard. The
    single-flight lock collapses pre-warm + concurrent first-user-request
    into one computation (P-6 belt-and-suspenders: invalidate→prewarm has
    no awaitable gap).
    """
    from __future__ import annotations
    from typing import Callable

    from loguru import logger
    from sqlalchemy.ext.asyncio import AsyncSession

    from localstock.cache import get_or_compute, resolve_latest_run_id
    from localstock.observability.metrics import get_metrics


    async def prewarm_hot_keys(
        session_factory: Callable[[], "AsyncSession"],
        *,
        ranking_limit: int = 50,
    ) -> None:
        """Pre-warm /scores/top (top {ranking_limit}) and /market/summary.

        Failure semantics (P-5): every error is caught + counted +
        logged, NEVER re-raised. Caller (`run_daily_pipeline`) treats
        pre-warm as best-effort.
        """
        run_id = await resolve_latest_run_id(session_factory)
        if run_id is None:
            logger.warning("cache.prewarm.skipped", reason="no_completed_run")
            return

        # --- /scores/top --------------------------------------------------
        try:
            from localstock.services.scoring_service import ScoringService

            async def _compute_ranking():
                async with session_factory() as session:
                    service = ScoringService(session)
                    stocks = await service.get_top_stocks(limit=ranking_limit)
                    return {"stocks": stocks, "count": len(stocks or [])}

            await get_or_compute(
                namespace="scores:ranking",
                key=f"limit={ranking_limit}:run={run_id}",
                compute_fn=_compute_ranking,
            )
            logger.info("cache.prewarm.ok", namespace="scores:ranking",
                        key=f"limit={ranking_limit}:run={run_id}")
        except Exception:
            logger.exception("automation.cache.prewarm_failed",
                             namespace="scores:ranking")
            try:
                get_metrics()["cache_prewarm_errors_total"].labels(
                    cache_name="scores:ranking"
                ).inc()
            except Exception:
                pass

        # --- /market/summary ---------------------------------------------
        try:
            # Mirror the route's _compute body. To avoid duplication, the
            # route module exposes its compute via a helper. For now, call
            # the underlying repository methods directly the same way the
            # route does. (Refactor opportunity: extract market_summary
            # builder into MarketService.get_summary — left as deferred.)
            from localstock.api.routes.market import build_market_summary
            async def _compute_market():
                async with session_factory() as session:
                    return await build_market_summary(session)

            await get_or_compute(
                namespace="market:summary",
                key=f"run={run_id}",
                compute_fn=_compute_market,
            )
            logger.info("cache.prewarm.ok", namespace="market:summary",
                        key=f"run={run_id}")
        except Exception:
            logger.exception("automation.cache.prewarm_failed",
                             namespace="market:summary")
            try:
                get_metrics()["cache_prewarm_errors_total"].labels(
                    cache_name="market:summary"
                ).inc()
            except Exception:
                pass

        logger.info("cache.prewarm.completed", run_id=run_id)
    ```
