    ```python
    """Phase 24 / OBS-17 — migration round-trip integration tests.

    Verifies that the 24a1b2c3d4e5 migration adds + removes the 4
    *_duration_ms columns cleanly, matching VALIDATION.md.
    """
    from __future__ import annotations

    import subprocess

    import pytest
    import sqlalchemy as sa

    from localstock.db.database import get_engine

    EXPECTED = {"crawl_duration_ms", "analyze_duration_ms",
                "score_duration_ms", "report_duration_ms"}


    async def _columns() -> set[str]:
        eng = get_engine()
        async with eng.begin() as conn:
            rows = await conn.execute(sa.text(
                "SELECT column_name FROM information_schema.columns "
                "WHERE table_name='pipeline_runs'"
            ))
            return {r[0] for r in rows}


    @pytest.mark.asyncio
    @pytest.mark.requires_pg
    async def test_migration_upgrade_adds_columns():
        # Assume conftest left DB at head; verify columns present.
        cols = await _columns()
        assert EXPECTED.issubset(cols), f"Missing: {EXPECTED - cols}"


    @pytest.mark.asyncio
    @pytest.mark.requires_pg
    async def test_migration_downgrade_removes_columns(tmp_path):
        """Run downgrade -1 then upgrade head; verify both states."""
        subprocess.run(["uv", "run", "alembic", "downgrade", "-1"],
                       cwd="apps/prometheus", check=True, capture_output=True)
        try:
            cols_after_down = await _columns()
            assert EXPECTED.isdisjoint(cols_after_down), \
                f"Columns not dropped: {EXPECTED & cols_after_down}"
        finally:
            subprocess.run(["uv", "run", "alembic", "upgrade", "head"],
                           cwd="apps/prometheus", check=True, capture_output=True)
        cols_after_up = await _columns()
        assert EXPECTED.issubset(cols_after_up)
    ```
