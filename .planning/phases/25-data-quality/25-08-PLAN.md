    ```python
    def _last_trading_day_on_or_before(today: date) -> date:
        """Return the most recent VN trading day on or before ``today``.

        Walks backwards skipping weekends and ``_VN_HOLIDAYS_2025_2026``.
        """
        d = today
        # Bound the loop defensively (worst-case Tet ~9 holidays + weekend).
        for _ in range(20):
            if _is_business_day(d):
                return d
            d = d - timedelta(days=1)
        # Fallback: if the calendar is exhausted (shouldn't happen) return as-is.
        return d
    ```

    **Step 2.** Replace the `health_data` endpoint body (lines 184–196):

    ```python
    @router.get("/health/data", summary="Data freshness — max price date vs calendar")
    async def health_data(session: AsyncSession = Depends(get_session)) -> dict:
        from localstock.config import get_settings  # local import: avoid module-load cost

        result = await session.execute(select(func.max(StockPrice.date)))
        max_date: date | None = result.scalar_one_or_none()
        today = date.today()
        last_trading = _last_trading_day_on_or_before(today)
        threshold = get_settings().dq_stale_threshold_sessions

        if max_date is None:
            # Cold start — preserve back-compat keys, mark unknown.
            return {
                "max_price_date": None,
                "trading_days_lag": None,
                "stale": True,
                "data_freshness": {
                    "last_trading_day": last_trading.isoformat(),
                    "max_data_date": None,
                    "sessions_behind": None,
                    "status": "unknown",
                    "threshold_sessions": threshold,
                },
            }

        lag = _trading_days_lag(max_date, today)
        # sessions_behind == lag computed against last_trading_day specifically;
        # _trading_days_lag already uses the calendar so the values agree.
        sessions_behind = lag
        if sessions_behind <= threshold:
            status = "fresh"
        else:
            status = "stale"

        return {
            "max_price_date": max_date.isoformat(),
            "trading_days_lag": lag,
            "stale": lag > 1,  # PRESERVED — back-compat with Phase 24 contract (D-05)
            "data_freshness": {
                "last_trading_day": last_trading.isoformat(),
                "max_data_date": max_date.isoformat(),
                "sessions_behind": sessions_behind,
                "status": status,
                "threshold_sessions": threshold,
            },
        }
    ```

    **Note on `stale` vs `status`:** the existing top-level `stale` flag uses
    a hard-coded `> 1` (Phase 24 behavior, preserved per D-05); the new
    `data_freshness.status` uses the configurable threshold. This duplication
    is intentional for back-compat; new clients should consume `status`.
