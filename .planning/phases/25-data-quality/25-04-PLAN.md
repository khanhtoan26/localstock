    ```python
    from localstock.dq import MAX_ERROR_CHARS
    from localstock.dq.sanitizer import sanitize_jsonb  # already imported in 25-02


    def _truncate_error(exc: BaseException) -> str:
        """Format an exception as '{Class}: {str(exc)[:MAX_ERROR_CHARS]}'.

        Used by per-symbol isolation wrappers (DQ-05 / 25-06) and by
        _write_stats when surfacing failed_symbols.error to JSONB.

        Per CONTEXT D-07 specifics + RESEARCH Pitfall G — bounds the size of
        the failed_symbols list so a 400-symbol pipeline cannot blow the
        stats JSONB into MB territory.
        """
        msg = str(exc)
        if len(msg) > MAX_ERROR_CHARS:
            msg = msg[:MAX_ERROR_CHARS] + "..."
        return f"{type(exc).__name__}: {msg}"
    ```

    ```python
        def _write_stats(
            self,
            run: PipelineRun,
            *,
            succeeded: int,
            failed: int,
            skipped: int,
            failed_symbols: list[dict],
        ) -> None:
            """Write PipelineRun.stats JSONB AND dual-write scalar columns (D-07).

            failed_symbols shape: [{"symbol": str, "step": str, "error": str}, ...]
            Errors are pre-truncated by callers via _truncate_error.

            Per CONTEXT D-07: dual-write to symbols_total/success/failed scalar
            columns through v1.5 for back-compat with automation_service +
            health_probe (RESEARCH §Audit List — Readers).
            """
            stats = sanitize_jsonb(
                {
                    "succeeded": succeeded,
                    "failed": failed,
                    "skipped": skipped,
                    "failed_symbols": failed_symbols,
                }
            )
            run.stats = stats
            # Dual-write scalars — through v1.5 (CONTEXT D-07)
            run.symbols_total = succeeded + failed + skipped
            run.symbols_success = succeeded
            run.symbols_failed = failed
            # Keep `errors` populated for back-compat with rescue path
            # (run_full lines 222-249 still consume it)
            if failed_symbols and run.errors is None:
                run.errors = {
                    "failed_symbols": [fs["symbol"] for fs in failed_symbols],
                }
    ```
