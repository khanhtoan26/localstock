    ```python
    """Phase 25 / DQ-03 — Per-rule Tier 2 mode dispatcher (D-06)."""
    from __future__ import annotations

    from typing import Literal

    from localstock.config import get_settings

    Mode = Literal["shadow", "enforce"]


    class Tier2Violation(Exception):
        """Raised by evaluate_tier2 when a Tier 2 rule is in 'enforce' mode."""

        def __init__(self, rule: str, offending) -> None:
            super().__init__(f"Tier 2 rule {rule!r} violated by {len(offending)} rows")
            self.rule = rule
            self.offending = offending


    def get_tier2_mode(rule_name: str) -> Mode:
        """Return mode for a given rule, falling back to default ('shadow').

        Per CONTEXT D-06: per-rule env flags `DQ_TIER2_<RULE>_MODE` override
        the global `DQ_DEFAULT_TIER2_MODE` default.
        """
        settings = get_settings()
        explicit = getattr(settings, f"dq_tier2_{rule_name}_mode", None)
        mode = (explicit or settings.dq_default_tier2_mode or "shadow").lower()
        if mode not in ("shadow", "enforce"):
            mode = "shadow"
        return mode  # type: ignore[return-value]
    ```

    ```python
    def evaluate_tier2(
        rule: str,
        df,
        predicate,
        *,
        symbol: str | None = None,
    ) -> None:
        """Run a Tier 2 advisory check.

        - `predicate(df) -> DataFrame of offending rows` (empty == pass).
        - Always emits `dq_warn` log + dq_violations_total counter.
        - In 'enforce' mode raises Tier2Violation; in 'shadow' returns None.

        Per CONTEXT D-06 + D-03 + RESEARCH Pattern 2.
        """
        from prometheus_client import REGISTRY
        from loguru import logger

        from localstock.dq.shadow import get_tier2_mode, Tier2Violation

        bad = predicate(df)
        if bad is None or (hasattr(bad, "empty") and bad.empty):
            return

        mode = get_tier2_mode(rule)
        tier = "advisory" if mode == "shadow" else "strict"

        try:
            counter = REGISTRY._names_to_collectors.get("localstock_dq_violations_total")
            if counter is not None:
                counter.labels(rule=rule, tier=tier).inc(len(bad))
        except Exception:
            logger.debug("dq.tier2.metric_lookup_failed", rule=rule)

        logger.warning(
            "dq_warn",
            rule=rule, tier=tier, symbol=symbol,
            violation_count=len(bad),
        )

        if mode == "enforce":
            raise Tier2Violation(rule, bad)
    ```

    ```python
    # ----------------------------------------------------------------------
    # Tier 2 (advisory) — RESEARCH §Pattern 2.
    # Used as predicates by evaluate_tier2 from AnalysisService.
    # ----------------------------------------------------------------------
    def predicate_gap_30pct(df):
        """Return rows where consecutive close-to-close gap exceeds 30%."""
        s = df.sort_values("date").assign(
            prev=lambda d: d["close"].shift(1),
            pct=lambda d: (d["close"] - d.get("prev", d["close"])).abs()
                          / d.get("prev", d["close"]).replace(0, float("nan")),
        )
        return s[s["pct"] > 0.30].dropna(subset=["pct"])


    def predicate_missing_rows_20pct(df, *, expected_session_count: int):
        """Return the rows-list-as-frame if missing > 20%, else empty."""
        actual = len(df)
        if expected_session_count <= 0:
            return df.iloc[0:0]
        missing_ratio = 1.0 - (actual / expected_session_count)
        if missing_ratio > 0.20:
            # Return at least one row so evaluate_tier2 sees a non-empty frame
            return df.iloc[0:1] if not df.empty else df
        return df.iloc[0:0]
    ```

    ```markdown
    # DQ Tier 2 Promotion Runbook

    > Phase 25 / DQ-03 (CONTEXT D-06). Operational procedure for promoting
    > a Tier 2 advisory rule from shadow → strict mode after the 14-day
    > shadow window per CONTEXT D-03.

    ## Why this runbook exists

    Phase 25 ships every Tier 2 rule (`rsi_anomaly`, `gap`, `missing_rows`)
    in **shadow mode** by default. Shadow mode emits the same Prometheus
    counter (`localstock_dq_violations_total{rule, tier="advisory"}`) and
    structured `dq_warn` logs as strict mode would, but **does NOT block**
    the pipeline. After a 14-day observation window we may promote a rule
    to strict mode (it then DOES block — failing rows are caught by the
    per-symbol try/except added in 25-06 / DQ-05 and surface in
    `PipelineRun.stats.failed_symbols`).

    ## Promotion Criteria (must ALL hold)

    1. Rule has been in `tier="advisory"` for ≥ 14 calendar days (CONTEXT D-03).
    2. 14-day violation rate (rows-flagged / rows-evaluated) < 5%.
    3. No false-positive review found in the past 7 days (manual triage).
    4. Pre-flight: confirm `localstock_dq_violations_total{rule=<name>, tier="strict"}`
       series exists in Prometheus (label exemplar on advisory side has been seen).
    5. AutomationService digest (Telegram daily) reviewed — no operator concern.

    ## Procedure: shadow → strict

    1. **Observe** — query Prometheus over last 14 days:
       ```
       sum(rate(localstock_dq_violations_total{rule="<name>", tier="advisory"}[14d]))
       /
       sum(rate(localstock_dq_validation_total{validator="<name>", outcome="pass"}[14d]))
       ```
       Confirm < 0.05.
    2. **Flip the env flag** in the deployed `.env` (production):
       ```
       DQ_TIER2_<NAME>_MODE=enforce
       ```
       (e.g., `DQ_TIER2_RSI_MODE=enforce`).
    3. **Restart** the API + scheduler so Settings re-loads.
       (`get_settings.cache_clear()` is called automatically on next request.)
    4. **Verify** — within 1 hour, Prometheus shows
       `localstock_dq_violations_total{rule="<name>", tier="strict"}` series
       starting to populate. The advisory series stops incrementing.
    5. **Watch the next pipeline run** — `PipelineRun.stats.failed_symbols`
       should show step="analyze" entries with error containing "Tier2Violation"
       for any rows that violate the rule.
    6. Update the **Per-Rule Status Table** (below) with the promotion date.

    ## Rollback (strict → shadow)

    If post-promotion the pipeline fails > 30% of symbols (Pitfall C echo —
    early hard-gate), roll back immediately:

    1. Set `DQ_TIER2_<NAME>_MODE=shadow` (or unset, default is shadow).
    2. Restart services.
    3. Confirm `tier="strict"` series stops growing.
    4. Open a debug ticket — investigate the false-positive surface before re-promoting.

    ## Per-Rule Status Table

    | Rule | Current Mode | Last Promoted | Notes |
    |------|--------------|---------------|-------|
    | `rsi_anomaly` | shadow | — | RSI > 99.5; pandas-ta computed |
    | `gap` | shadow | — | close-to-close > 30% (split adjustment may inflate) |
    | `missing_rows` | shadow | — | > 20% of expected sessions missing |

    - 25-RESEARCH.md §Pattern 2 + §Pitfall C (Tier 2 hard-gate)
    - Prometheus query examples in `docs/runbook/observability-queries.md`
    ```
