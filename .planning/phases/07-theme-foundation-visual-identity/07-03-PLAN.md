**File 1: `apps/helios/src/components/stock/ai-report-panel.tsx`**

```tsx
"use client";

import Markdown from "react-markdown";
import { ScrollArea } from "@/components/ui/scroll-area";
import { Skeleton } from "@/components/ui/skeleton";
import type { StockReport } from "@/lib/types";

interface AIReportPanelProps {
  report: StockReport | undefined;
  isLoading: boolean;
  isError: boolean;
}

export function AIReportPanel({ report, isLoading, isError }: AIReportPanelProps) {
  if (isLoading) {
    return (
      <div className="space-y-3">
        <Skeleton className="h-6 w-48" />
        <Skeleton className="h-4 w-full" />
        <Skeleton className="h-4 w-full" />
        <Skeleton className="h-4 w-3/4" />
        <Skeleton className="h-4 w-5/6" />
        <Skeleton className="h-4 w-full" />
        <Skeleton className="h-4 w-2/3" />
      </div>
    );
  }

  if (isError || !report) {
    return (
      <p className="text-sm text-muted-foreground">
        Chưa có báo cáo — chạy pipeline để tạo báo cáo AI.
      </p>
    );
  }

  const content = report.summary || null;
  const fallbackJson = report.content_json;

  return (
    <ScrollArea className="max-h-[calc(100vh-12rem)]">
      {content ? (
        <Markdown className="prose dark:prose-invert prose-sm max-w-none">
          {content}
        </Markdown>
      ) : fallbackJson ? (
        <div className="space-y-4">
          {Object.entries(fallbackJson).map(([key, value]) => (
            <div key={key}>
              <h3 className="text-sm font-semibold capitalize mb-1">
                {key.replace(/_/g, " ")}
              </h3>
              <p className="text-sm text-muted-foreground whitespace-pre-wrap">
                {typeof value === "string" ? value : JSON.stringify(value, null, 2)}
              </p>
            </div>
          ))}
        </div>
      ) : (
        <p className="text-sm text-muted-foreground">
          Báo cáo không có nội dung.
        </p>
      )}
    </ScrollArea>
  );
}
```

Key implementation details per user decisions:
- **D-17:** Uses `react-markdown` with `prose dark:prose-invert prose-sm max-w-none` classes. The `prose-sm` matches the existing 14px body text. `max-w-none` prevents prose from constraining width (the parent container handles width). `dark:prose-invert` handles dark theme.
- **D-18:** Wraps content in `ScrollArea` with `max-h-[calc(100vh-12rem)]` — gives the report ~80% of viewport height minus header/padding, with internal scroll via ScrollArea.
- **D-19:** Checks `report.summary` first → renders as markdown. If null/empty → formats `content_json` entries as sections with key-as-heading and value-as-paragraph. If both null → shows empty message.

**File 2: `apps/helios/src/components/stock/score-breakdown.tsx`**

```tsx
import { cn, formatScore } from "@/lib/utils";
import type { StockScore } from "@/lib/types";

const DIMENSIONS = [
  { key: "technical_score" as const, label: "Kỹ Thuật", color: "bg-blue-500" },
  { key: "fundamental_score" as const, label: "Cơ Bản", color: "bg-emerald-500" },
  { key: "sentiment_score" as const, label: "Tin Tức", color: "bg-amber-500" },
  { key: "macro_score" as const, label: "Vĩ Mô", color: "bg-violet-500" },
] as const;

interface ScoreBreakdownProps {
  score: StockScore;
}

export function ScoreBreakdown({ score }: ScoreBreakdownProps) {
  return (
    <div className="space-y-4">
      {DIMENSIONS.map((dim) => {
        const value = score[dim.key];
        const percent = value != null ? Math.min(Math.max(value, 0), 100) : 0;
        return (
          <div key={dim.key}>
            <div className="flex items-center justify-between mb-1">
              <span className="text-sm font-medium">{dim.label}</span>
              <span className="text-sm font-mono text-muted-foreground">
                {formatScore(value)}
              </span>
            </div>
            <div className="h-2 w-full rounded-full bg-muted">
              <div
                className={cn("h-2 rounded-full transition-all", dim.color)}
                style={{ width: `${percent}%` }}
              />
            </div>
          </div>
        );
      })}
    </div>
  );
}
```

Key implementation details per D-24:
- Shows 4 dimensions: Kỹ Thuật (technical_score), Cơ Bản (fundamental_score), Tin Tức (sentiment_score), Vĩ Mô (macro_score)
- Each dimension: label left-aligned, formatted score right-aligned (using `formatScore` from utils), visual progress bar below
- Progress bar: `bg-muted` track with colored fill bar, width = score clamped to 0-100%
- Distinct colors per dimension: blue, emerald, amber, violet — all visible on both themes
- No `"use client"` needed — this is a pure presentational component with no hooks/state
